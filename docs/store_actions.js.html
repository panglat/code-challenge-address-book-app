/**
 * @file Manages the configuration settings for the widget.
 * @module actions
 */
import {
  USERS_FETCH_REQUESTED,
  USERS_FETCH_SUCCEEDED,
  USERS_FETCH_FAILED,
  SET_USERS_RECORD_TO_DISPLAY,
  RESET_USERS_FETCH_FAILED,
  SET_USERS_SEARCH,
  SET_NATIONALITY_SEARCH,
  CLEAR_USERS_LIST,
} from './actionTypes';

/**
 * @function
 * @name requestUsers get user list action
 * @param {getUserListParams} params getUserList parameters.
 */
export const requestUsers = (params) => ({
  type: USERS_FETCH_REQUESTED,
  payload: { params },
});

/**
 * @function
 * @name requestUsersSuccess get user list success action
 * @param {response} response getUserList parameters.
 */
export const requestUsersSuccess = (response) => ({
  type: USERS_FETCH_SUCCEEDED,
  payload: { response },
});

/**
 * @function
 * @name requestUsersError get user list failed action
 * @param {error} error request error.
 */
export const requestUsersError = (error) => ({
  type: USERS_FETCH_FAILED,
  payload: { error },
});

/**
 * @function
 * @name setUsersRecordsToDisplay set number of records to display in the user list
 * @param {number} records number of records
 */
export const setUsersRecordsToDisplay = (records) => ({
  type: SET_USERS_RECORD_TO_DISPLAY,
  payload: { records },
});

/**
 * @function
 * @name resetUsersError reset user request error
 */
export const resetUsersError = () => ({
  type: RESET_USERS_FETCH_FAILED,
  payload: {},
});

/**
 * @function
 * @name clearUsersList clear user list
 */
export const clearUsersList = () => ({
  type: CLEAR_USERS_LIST,
  payload: {},
});

/**
 * @function
 * @name setUserSearch set user search
 * @param {string} userSearch user search
 */
export const setUserSearch = (userSearch) => ({
  type: SET_USERS_SEARCH,
  payload: { userSearch },
});

/**
 * @function
 * @name setNationalitySearch set nationality comma-separated values
 * @param {string} nationalitySearch comma-separated values
 */
export const setNationalitySearch = (nationalitySearch) => ({
  type: SET_NATIONALITY_SEARCH,
  payload: { nationalitySearch },
});
